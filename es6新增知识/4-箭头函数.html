es6普通函数：
		function show(a, b=100){  //默认b=100
			return a + b;
		}
		alert(show(1,2));  //3		
		alert(show(1));	   //101
	
	es6箭头函数：
		let show = (a,b=100) => a+b;//一个返回值时
		alert(show(1,2));  //3		
		alert(show(1));	   //101
		***************************************************
		let show = (a,b=100) => {//多个返回值时
			alert(`其他返回值`);
			return a+b;
		};
		alert(show(1,2));  //3
