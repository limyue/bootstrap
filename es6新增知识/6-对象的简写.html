es5对象：
		var a = 10;//定义一个变量
		var obj = {//定义一个对象
			a:a，   //将变量a复制给对象a
			b:function(){
				console.log(this.a)
			}

		}
	console.log(obj);
	obj.b();
	
	es6对象：
	var a = 10;
	var obj = {
		a      //es6中不用再赋值给键，直接引用即可
	}
	console.log(obj);

	var a = 10;

	es6将某一对象属性加到另外一个对象上面--Object.assign()
	var a = 10;
	var obj = {//定义一个对象
		a,
		b(){
			console.log(this.a)
		}
	}
	var obj2 = {
		c:100
	}
	Object.assign(obj2,obj);
	console.log(obj2)
