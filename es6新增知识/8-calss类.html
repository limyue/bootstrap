	function Manager(name,age,password){

	es6 引入class类的概念：
	class Person{
		//构造函数constructor(){}
		constructor(name,age){  
			this.name = name;
			this.age = age;
		}
		showPeople(){
			alert(`姓名${this.name},年龄${this.age}`)
		}
		//静态方法
		static getMethod(){
			return "hello";
		}
	}
	//var Person1 = new Person("li","11")
	//Person1.showPeople()
	//Person1.getMethod()  //报错--静态方法不能别重新构造
	console.log(Person.getMethod())

	//继承extends
	class Manager extends Person{
		constructor(name,age,password){
			super(name,age);
			this.password = password;
		}
	}
	var Manager1 = new Manager("yue","1","1234");
	//Manager1.showPeople();
